// Inisialisasi repositori
// Digunakan untuk membuat repositori Git baru di direktori lokal
git init

// Tambahkan file ke staging area
// Digunakan untuk menambahkan file tertentu ke staging area
git add < file >

// Commit perubahan
// Digunakan untuk menyimpan snapshot dari perubahan yang telah dilakukan pada repositori
git commit - m "pesan commit"

// Lihat status repositori
// Digunakan untuk melihat status file dalam repositori
git status

// Buat branch baru
// Digunakan untuk membuat branch baru dalam repositori
git branch < nama - branch >

// Pindah ke branch lain
// Digunakan untuk berpindah ke branch lain dalam repositori
git checkout < nama - branch >

// Gabungkan branch dengan branch aktif
// Digunakan untuk menggabungkan perubahan dari branch lain ke branch aktif
git merge < nama - branch >

// Tambahkan repo remote
// Digunakan untuk menambahkan remote repository
git remote add origin < url - repo >

// Kirim perubahan ke repo remote
// Digunakan untuk mengirim perubahan lokal ke repo remote
git push origin < nama - branch >

// Tarik perubahan terbaru dari repo remote
// Digunakan untuk menarik perubahan terbaru dari repo remote
git pull origin < nama - branch >

// Buat pull request
// Digunakan untuk membuat permintaan penggabungan perubahan dari branch ke branch utama
git pull - request

// Terima pull request
// Digunakan untuk menerima permintaan penggabungan yang diajukan oleh orang lain
git accept - pull - request

// Fork repositori
// Digunakan untuk membuat salinan independen dari repositori milik orang lain ke akun GitHub Anda
git fork

// Berikan bintang pada repositori
// Digunakan untuk memberikan bintang pada repositori yang menarik atau berguna
git star

// Lihat riwayat kontribusi
// Digunakan untuk melihat riwayat kontribusi yang Anda dan orang lain buat pada repositori
git contribution - history

// Kelola issues dan projects
// Digunakan untuk mengelola masalah (issues) dan proyek (projects) dalam repositori
git manage - issues - projects
